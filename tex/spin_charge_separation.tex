\[V = g\rho_1\rho_2 \rightarrow V= g_{||}(\rho_{1\uparrow}\rho_{2\uparrow}+\rho_{1\downarrow}\rho_{2\downarrow}) + g_\perp(\rho_{1\uparrow}\rho_{2\downarrow}+\rho_{1\downarrow}\rho_{2\uparrow})\]

\begin{align*}

g_\perp = g_\rho-g_\sigma

\begin{array}{@{}l}
